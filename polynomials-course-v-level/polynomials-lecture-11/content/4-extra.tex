\begin{section-problem}
    Probar que para todo $n$ entero positivo, se cumple que
    \[1 + \frac{1}{2} \cdot \binom{n + 1}{1}  + \frac{1}{4} \cdot \binom{n + 2}{2}  + \cdots + \frac{1}{2^{n - 1}} \cdot \binom{2n - 1}{n - 1} + \frac{1}{2^n} \cdot \binom{2n}{n} = 2^n\]
\end{section-problem}
