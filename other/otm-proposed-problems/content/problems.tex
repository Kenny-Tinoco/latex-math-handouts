\section{\large Problemas}

\begin{section-problem}
    Sabiedo que $S(x)$ es un polinomio cúbico mónico con $S(-10) = 1$, $S(-17) = 2$ y $S(-24) = 3$.
    Hallar el valor de $S(-3) - 35$.
\end{section-problem}


\begin{section-problem}
    Sean $a$, $b$ y $c$ las raíces del polinomio $6x^3 + x + 4046$.
    Calcular
    \[
        (a + b)^3 + (b + c)^3 + (c + a)^3.
    \]
\end{section-problem}
