\section{\large Exercises}

\begin{exer}
    Maximize $f(x,y) = x + 6y$. $x \geq 0$; $y \geq 0$; $x + 4y \leq 8$; $x + y \leq 5$.
\end{exer}

\begin{exer}
    Solve the system
    \[
        \begin{cases}
            3x - y + 2z = 15\\
            x - 2y - z = -5\\
            5x + 2y + z = 5
        \end{cases}
    \]
\end{exer}


\begin{exer}
    Solve for $X$
    \begin{equation*}
        \begin{pmatrix}
            2 & -1\\
            2 & -5\\
            -1 & 3
        \end{pmatrix} - 3X =
        \begin{pmatrix}
            1 & 7\\
            -4 & 8\\
            -3 & 1
        \end{pmatrix}
        \times
        \begin{pmatrix}
            6 & -7\\
            2 & -3\\
        \end{pmatrix}
    \end{equation*}
\end{exer}
